WITH RECURSIVE
    dates (date) AS (
        VALUES
            ('2015-10-03')
        UNION ALL
        SELECT
            date(date, '+1 day')
        FROM
            dates
        WHERE
            date < '2015-11-01'
    )
SELECT
    date
FROM
    dates;
